  - path: /etc/ssl/openssl.cnf
    owner: root:root
    permissions: '0644'
    append: true
    content: |
      # CCN-STIC-610 policy (auto-managed)
      MinProtocol = TLSv1.2
      CipherString = DEFAULT:@SECLEVEL=2

  # Script de escaneo diario
  - path: /opt/clamav/clamav_scan.sh
    owner: root:root
    permissions: '0755'
    content: |
      #!/bin/bash
      LOGFILE="/var/log/clamav/scan-$(date +%F).log"
      if command -v clamdscan >/dev/null 2>&1; then
        clamdscan -l "$LOGFILE" --fdpass --exclude-dir="^/sys" --exclude-dir="^/proc" --exclude-dir="^/dev" /
      else
        clamscan -r -i --exclude-dir="^/sys" --exclude-dir="^/proc" --exclude-dir="^/dev" / > "$LOGFILE"
      fi
      find /var/log/clamav/ -type f -name "scan-*.log" -mtime +15 -delete 2>/dev/null || true

  # Unidad systemd del servicio de escaneo
  - path: /etc/systemd/system/clamav-scan.service
    owner: root:root
    permissions: '0644'
    content: |
      [Unit]
      Description=Escaneo antivirus diario con ClamAV

      [Service]
      Type=oneshot
      ExecStart=/opt/clamav/clamav_scan.sh

  # Timer diario
  - path: /etc/systemd/system/clamav-scan.timer
    owner: root:root
    permissions: '0644'
    content: |
      [Unit]
      Description=Temporizador diario para ClamAV

      [Timer]
      OnCalendar=daily
      Persistent=true

      [Install]
      WantedBy=timers.target

  - systemctl enable --now clamav-scan.timer

      deny = 3
      unlock_time = 0
      fail_interval = 1800
      # even_deny_root deshabilitado (seguro)
      admin_group = wheel

  - touch /etc/security/faillock.conf

  # Resetear bloqueos actuales
  - |
      if command -v faillock >/dev/null 2>&1; then
        faillock --reset || true
        faillock --user root --reset || true
      fi

  # Limpiar directorio de bloqueos
  - rm -rf /run/faillock/* || true

  # ComprobaciГіn opcional de presencia de pam_faillock (no modifica nada)
  - |
      for f in /etc/pam.d/system-login /etc/pam.d/system-local-login /etc/pam.d/system-remote-login; do
        [ -f "$f" ] || continue
        grep -q 'pam_faillock\.so' "$f" || echo "Advertencia: $f no contiene pam_faillock"
      done

  - usbguard

  # Endurecimiento OpenSSL
  - path: /etc/ssl/openssl.cnf
    owner: root:root
    permissions: '0644'
    append: true
    content: |
      # CCN-STIC-610 - Endurecimiento OpenSSL
      [system_default_sect]
      MinProtocol = TLSv1.2
      CipherString = DEFAULT@SECLEVEL=2

  # Endurecimiento GnuTLS
  - path: /etc/gnutls/config
    owner: root:root
    permissions: '0644'
    content: |
      system_priority = NORMAL:-VERS-SSL3.0:-VERS-TLS1.0:-VERS-TLS1.1:-SIGN-SHA1:+AEAD

  # ConfiguraciГіn del daemon de usbguard
  - path: /etc/usbguard/usbguard-daemon.conf
    owner: root:root
    permissions: '0644'
    content: |
      # CCN-STIC-610 - ProtecciГіn de Dispositivos USB
      RuleFile=/etc/usbguard/rules.conf
      ImplicitPolicyTarget=block

      PresentDevicePolicy=keep
      PresentControllerPolicy=apply-policy
      InsertedDevicePolicy=apply-policy

      RestoreControllerDeviceState=false
      DeviceManagerBackend=uevent

      IPCAllowedUsers=root
      IPCAllowedGroups=sudo
      IPCAccessControlFiles=/etc/usbguard/IPCAccessControl.d/

      DeviceRulesWithPort=false

      AuditBackend=FileAudit
      AuditFilePath=/var/log/usbguard/usbguard-audit.log

runcmd:
  # Crear directorios necesarios
  - mkdir -p /etc/usbguard/IPCAccessControl.d
  - mkdir -p /var/log/usbguard

  # Generar reglas iniciales si no existen
  - |
      if [ ! -s /etc/usbguard/rules.conf ]; then
        usbguard generate-policy --insert-policy=allow > /etc/usbguard/rules.conf || exit 1
      fi

  # Activar y reiniciar usbguard
  - systemctl enable --now usbguard.service
  - systemctl restart usbguard.service

##### FILE: ./kernel/configs/base/50-hardening-crypto.yaml #####

#cloud-config

write_files: []

runcmd:
  - |
      #!/bin/bash
      set -euo pipefail

      SECLVL=${SECLVL:-2}
      MIN_TLS=${MIN_TLS:-TLSv1.2}
      GNUTLS_PROFILE=${GNUTLS_PROFILE:-"SECURE128:-VERS-SSL3.0:-VERS-TLS1.0:-VERS-TLS1.1"}

      OPENSSL_CNF=/etc/ssl/openssl.cnf
      GNUTLS_PRI=/etc/gnutls/default-priorities
      SSHD_CFG=/etc/ssh/sshd_config

      # --- OpenSSL hardening ---
      if [ -f "$OPENSSL_CNF" ]; then
        cp -f "$OPENSSL_CNF" "${OPENSSL_CNF}.bak_$(date +%F_%H%M%S)" || true

        if ! grep -q '^openssl_conf\s*=\s*default_conf' "$OPENSSL_CNF" 2>/dev/null; then
          printf '\nopenssl_conf = default_conf\n' >> "$OPENSSL_CNF"
        fi
        if ! grep -q '^

\[ *default_conf *\]

' "$OPENSSL_CNF"; then
          printf '\n[ default_conf ]\nssl_conf = ssl_sect\n' >> "$OPENSSL_CNF"
        elif ! grep -q '^ssl_conf\s*=\s*ssl_sect' "$OPENSSL_CNF"; then
          sed -i -E 's/^

\[ *default_conf *\]

.*/[ default_conf ]\nssl_conf = ssl_sect/' "$OPENSSL_CNF"
        fi
        if ! grep -q '^

\[ *ssl_sect *\]

' "$OPENSSL_CNF"; then
          printf '\n[ ssl_sect ]\nsystem_default = system_default_sect\n' >> "$OPENSSL_CNF"
        elif ! grep -q '^system_default\s*=\s*system_default_sect' "$OPENSSL_CNF"; then
          sed -i -E 's/^

\[ *ssl_sect *\]

.*/[ ssl_sect ]\nsystem_default = system_default_sect/' "$OPENSSL_CNF"
        fi

        if grep -q '^

\[ *system_default_sect *\]

' "$OPENSSL_CNF"; then
          sed -i -E "\|^

\[ *system_default_sect *\]

|,$ s/^MinProtocol.*/MinProtocol = ${MIN_TLS}/" "$OPENSSL_CNF" || true
          sed -i -E "\|^

\[ *system_default_sect *\]

|,$ s/^CipherString.*/CipherString = DEFAULT@SECLEVEL=${SECLVL}/" "$OPENSSL_CNF" || true
          awk -v mt="${MIN_TLS}" -v slv="${SECLVL}" '
            BEGIN{insec=0; min=0; cs=0}
            /^

\[ *system_default_sect *\]

/{print; insec=1; next}
            insec==1 && /^MinProtocol/{min=1}
            insec==1 && /^CipherString/{cs=1}
            {print}
            END{
              if(!min || !cs){
                print "MinProtocol = " mt
                print "CipherString = DEFAULT@SECLEVEL=" slv
              }
            }
          ' "$OPENSSL_CNF" >"$OPENSSL_CNF.tmp" && mv "$OPENSSL_CNF.tmp" "$OPENSSL_CNF"
        else
          cat >> "$OPENSSL_CNF" <<EOF

[ system_default_sect ]
MinProtocol = ${MIN_TLS}
CipherString = DEFAULT@SECLEVEL=${SECLVL}
EOF
        fi
      fi

      # --- GnuTLS hardening ---
      install -d -m 0755 "$(dirname "$GNUTLS_PRI")"
      cp -f "$GNUTLS_PRI" "${GNUTLS_PRI}.bak_$(date +%F_%H%M%S)" 2>/dev/null || true
      printf '%s\n' "$GNUTLS_PROFILE" > "$GNUTLS_PRI"
      chmod 0644 "$GNUTLS_PRI"

      # --- OpenSSH hardening ---
      if [ -f "$SSHD_CFG" ]; then
        cp -f "$SSHD_CFG" "${SSHD_CFG}.bak_$(date +%F_%H%M%S)" || true

        ensure() {
          local key="$1" val="$2"
          if grep -qE "^#?${key} " "$SSHD_CFG"; then
            sed -i -E "s/^#?${key} .*/${key} ${val}/" "$SSHD_CFG"
          else
            echo "${key} ${val}" >> "$SSHD_CFG"
          fi
        }

        ensure KexAlgorithms 'curve25519-sha256,curve25519-sha256@libssh.org'
        ensure Ciphers 'chacha20-poly1305@openssh.com,aes256-gcm@openssh.com,aes256-ctr'
        ensure MACs 'hmac-sha2-512-etm@openssh.com,hmac-sha2-256-etm@openssh.com'
        ensure CASignatureAlgorithms 'rsa-sha2-512,rsa-sha2-256'

        if sshd -t; then
          systemctl restart ssh || systemctl restart sshd || exit 1
        else
          exit 1
        fi
      fi
